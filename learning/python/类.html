# 定义

类定义通过 class 关键字。class 后面紧跟类名，类名通常大写开头。紧接着是 (object) ，表示定义的类从 oject 继承来的。特殊的 init 方法为构造函数。init 第一个参数永远为 self，表示实例本身。

class Student(object):
 def __init__(self, name):
        self.name = name

# 访问限制

如果要让内部属性不被外部访问，可以把属性的名称前加上两个下划线 __ ，在Python中，实例的变量名如果以 __ 开头，就变成了一个私有变量（private），只有内部可以访问，外部不能访问.

# 继承和多态

class Animal(object):
    def run(self):
        print('animal fun')
    # 静态属性
    name = 123

class Dog(Animal):
    def __init__(self):
        # 实例属性
        self.name = 666
    def show(self):
        print(self.name)
# 打印 666
Dog().show()

# 修改实例属性和方法

from types import MethodType

def say(self):
    print(self.name)
class A(object):
    pass
a = A()
a.name = 'haha'
# 不绑定 self(属性)
a.say = say
a.say(a)
# 绑定 self
a._say = MethodType(say,a)
a._say()

# 动态绑定类方法

class A(object):

a = A()
a.name = 'haha'
def sayHello(self):

A.sayHello = sayHello
a.sayHello()

# slot 限制实例属性扩展

from types import MethodType
class A(object):
    __slots__ = ("name")
    pass
a = A()
a.name = "aa"
def say(self):
    print(self.name)
# 报错
# a.say = say
# a._say = MethodType(say,a)
A.say = say
a.say()

# 对实例属性读写控制

使用 @property 和 property.setter 对属性读写控制

class A(object):
    def __init__(self, name):
        self._name = name
    @property
    def name(self):
        return self._name
    @name.setter
    def name(self, name):
        self._name = 'set_' + name

a = A('zzz')
a.name = 'lam'
print(a.name)

# 定制类

  1. __str__
class A():
    def __str__(self):
        return 'a'
a = A()
c = str(a)
  1. __iter__：如果一个类想被用于 for ... in 循环，类似 list 或 tuple 那样，就必须实现一个 __iter__ 方法，该方法返回一个迭代对象，然后，Python的for循环就会不断调用该迭代对象的 __next__ 方法拿到循环的下一个值，直到遇到StopIteration错误时退出循环。
class A():
    def __init__(self):
        self.count = 0
    def __iter__(self):
        return self
    def __next__(self):
        if(self.count>10):
            raise StopIteration()
        self.count+=1
        return self.count
        
for item in A():
    print(item)
  1. __getitem__：
class A():
    def __init__(self):
        self.count = 0

    def __getitem__(self, n):
        a, b = 1, 1
        # a[2]
        if isinstance(n, int):
            for x in range(n):
                a, b = b, a + b
            return a
        # a[2, 5]
        if isinstance(n, slice):
            l1 = []
            start = n.start
            stop = n.stop
            if start == None:
                start = 0
            for i in range(stop):
                if i >= start:
                    l1.append(a)
                a, b = b, a + b

            return l1
  1. __getattr__：当访问不存在的实例属性时，会调用getattr
class A():
    def __init__(self):
        self.count = 0

    def __getattr__(self, attr):
        return attr
  1. __call__：使实例可以直接被调用
class A():

        self.count = 0

    def __call__(self,):
       print(self)

a = A()

# 枚举类

from enum import Enum, unique
Weeks = Enum('Weeks', ('Mon', 'Tues', 'Wed'))

# Mon Weeks.Mon 1
# Tues Weeks.Tues 2
# Wed Weeks.Wed 3
for name, member in Weeks.__members__.items():
    print(name, member, member.value)

@unique
class Month(Enum):
    Jan = 11
    Feb = 22

# Jan Month.Jan 11
# Feb Month.Feb 22
for name, member in Month.__members__.items():
    print(name, member, member.value)

# 元类

class A():
    pass
a = A()

type 函数可以查看一个类型或变量的类型，A 是一个 class，它的类型就是 type，而a是一个实例，它的类型就是class A

type 函数既可以返回一个对象的类型，又可以创建出新的类型，比如，我们可以通过 type 函数创建出类，而无需通过 class A(object)... 的定义：

def say(self):
    print(self.name)
# 打印 wode
A = type('Hello', (object,), {'name': 'wode', 'say': say})
a = A()
a.say()

class ListMetaclass(type):
    def __new__(cls, name, bases, attrs):
        attrs['add'] = lambda self, value: self.append(value)
        return type.__new__(cls, name, bases, attrs)
class A(list, metaclass=ListMetaclass):
    pass
a = A()
a.add(1111)
